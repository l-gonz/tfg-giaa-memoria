\newacronym{uav}{UAV}{Unmanned Aerial Vehicle}

\newglossaryentry{mavlink}{%
  name={MAVlink},
  description={A lightweight messaging protocol developed by the Dronecode Project for communicating with drones and between onboard drone components.}
}

\newglossaryentry{sitl}{%
  name={SITL},
  description={Software-in-the-loop is a simulation mode where the flight stack runs on a standard computer instead of a flight controller.}
}

\newglossaryentry{hitl}{%
  name={HITL},
  description={Hardware-in-the-Loop (HITL or HIL) is a simulation mode in which normal PX4 firmware is run on real flight controller hardware, while all required sensors are simulated.}
}

\newglossaryentry{gpio}{%
  name={GPIO},
  description={General Purpose Input Output (Pins)}
}

\newglossaryentry{udp}{%
  name={UDP},
  description={User Datagram Protocol}
}
